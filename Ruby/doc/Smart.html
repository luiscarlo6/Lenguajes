def next m
  if m==nil 
    @contNil +=1
  end
  
  if @contNil < 2
    if m==nil
      return Scissors.new
    end
    sumar(m)
    num = rand(@p + @r + @s -1).to_int
    if (0 <= num) && (num < @p)
      Scissors.new
    elsif (@p <= num) && (num < @p+@r)
      Paper.new
    elsif (@p+@r <= num) && (num < @p+@r+@s)
      Rock.new
    end
  end
end

def sumar m
  case m
  when Paper
    @p += 1
  when Rock
    @r += 1
  when Scissors
    @s += 1
  end
end
