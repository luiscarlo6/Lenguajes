def next m
  if m==nil
    @contNil +=1
  end
  
  if @contNil <2
    if m == nil
      return @estrategia
    end
    
    if @control
      @control = false
      @anterior = m
      @estrategia
    else
      @estrategia = @anterior
      @anterior = m
      @estrategia
    end
  end
end
